custom_block_command.rb

The CustomBlockCommand class is reserved for block commands which have been imported into the Papyrus lexicon using a CustomCommandSet. It is one of three possible node types which the Parser produces.

    def _evaluates_as_command_or_variable?
      _evaluates_as_custom_block_command? or _evaluates_as_variable?
    end

    def _evaluates_as_custom_block_command?
      exec_command = (
        _command_allowed? &&
        self.template.custom_commands.class.has_block_command?(@name)
      )
      if exec_command
        @result = self.template.custom_commands.__call_block_command__(self)
        return true
      else
        return false
      end
    rescue ParserError => e
      return false
    end
